// Stop ingredient erin
function versliert(ingredient) {
  switch (ingredient) {
    case 'vlees':
      return 'pasta'; // Maak gehakt
    case 'pastadeeg':
      return 'spaghetti'; // Maak spaghetti
    case 'rijstbloem':
      return 'mihoen'; // Maak mihoen
  }
}

De test


// Module
describe('Verslierteraar', function () {

  // Onderdeel
  context('pastadeeg', function () {

    // Voer de test uit
    expect(versliert('pastadeeg'))
      .to.be.a('string')        // Resultaat moet een 'string' zijn
      .and.equal('spaghetti');  // Resultaat moet 'spahetti' zijn

  });

});
